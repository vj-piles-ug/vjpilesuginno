"use client"

import { Suspense } from "react"
import { useAuth } from "@/lib/auth-context"
import { useRouter } from "next/navigation"
import { useEffect, useState } from "react"
import AdminNav from "@/components/admin-nav"
import AdminMobileNav from "@/components/admin-mobile-nav"
import { Card } from "@/components/ui/card"
import { database } from "@/lib/firebase"
import { ref, onValue, remove } from "firebase/database"
import {
  Wallet,
  CheckCircle2,
  XCircle,
  Clock,
  Phone,
  Mail,
  User,
  Search,
  ArrowUpDown,
  Filter,
  Trash2,
} from "lucide-react"

export default function WalletPageWrapper() {
  return (
    <Suspense fallback={<div className="min-h-screen bg-background flex items-center justify-center"><div className="w-8 h-8 border-2 border-primary border-t-transparent rounded-full animate-spin" /></div>}>
      <AdminWalletPage />
    </Suspense>
  )
}

interface PaymentLog {
  id: string
  userId: string
  email: string
  phoneNumber: string
  firstName: string
  lastName: string
  amount: number
  planId: string
  planName: string
  days: number
  orderTrackingId: string
  merchantReference: string
  status: string
  paymentMethod?: string
  confirmationCode?: string
  createdAt: string
  updatedAt?: string
}

function AdminWalletPage() {
  const { user, loading, isAdmin } = useAuth()
  const router = useRouter()
  const [payments, setPayments] = useState<PaymentLog[]>([])
  const [filteredPayments, setFilteredPayments] = useState<PaymentLog[]>([])
  const [loadingData, setLoadingData] = useState(true)
  const [searchQuery, setSearchQuery] = useState("")
  const [statusFilter, setStatusFilter] = useState<"all" | "successful" | "failed" | "pending">("all")
  const [sortOrder, setSortOrder] = useState<"newest" | "oldest">("newest")

  useEffect(() => {
    if (!loading && (!user || !isAdmin)) {
      router.push("/login")
    }
  }, [user, loading, isAdmin, router])

  function normalizeStatus(status: string | undefined): string {
    if (!status) return "pending"
    const s = status.toLowerCase()
    if (s === "completed" || s === "successful" || s === "success") return "successful"
    if (s === "failed" || s === "fail") return "failed"
    if (s === "reversed") return "reversed"
    return "pending"
  }

  // Real-time Firebase listener - transactions appear instantly when written
  useEffect(() => {
    if (!isAdmin) return

    setLoadingData(true)

    // Listen to paymentLogs in real-time
    const logsRef = ref(database, "paymentLogs")
    const subsRef = ref(database, "subscriptions")

    let logsData: any = null
    let subsData: any = null
    let gotLogs = false
    let gotSubs = false

    const buildPayments = () => {
      if (!gotLogs || !gotSubs) return

      const paymentsMap: Record<string, PaymentLog> = {}

      // 1. Payment logs (primary source)
      if (logsData && typeof logsData === "object") {
        for (const [key, value] of Object.entries(logsData)) {
          const v = value as any
          paymentsMap[key] = {
            id: key,
            userId: v.userId || "",
            email: v.email || "",
            phoneNumber: v.phoneNumber || v.phone_number || v.payment_account || "",
            firstName: v.firstName || v.first_name || "",
            lastName: v.lastName || v.last_name || "",
            amount: v.amount || 0,
            planId: v.planId || "",
            planName: v.planName || "",
            days: v.days || 0,
            orderTrackingId: v.orderTrackingId || key,
            merchantReference: v.merchantReference || "",
            status: normalizeStatus(v.status),
            paymentMethod: v.paymentMethod || v.payment_method || "",
            confirmationCode: v.confirmationCode || v.confirmation_code || "",
            createdAt: v.createdAt || v.created_at || "",
            updatedAt: v.updatedAt || v.updated_at || "",
          }
        }
      }

      // 2. Subscriptions - fill in any not already in paymentLogs
      if (subsData && typeof subsData === "object") {
        for (const [userId, value] of Object.entries(subsData)) {
          const v = value as any
          const trackingId = v.orderTrackingId || `sub-${userId}`
          if (!paymentsMap[trackingId]) {
            paymentsMap[trackingId] = {
              id: trackingId,
              userId,
              email: v.email || "",
              phoneNumber: v.paymentPhone || v.phoneNumber || "",
              firstName: v.firstName || v.displayName || "",
              lastName: v.lastName || "",
              amount: v.amount || 0,
              planId: v.planId || "",
              planName: v.planName || "",
              days: v.days || 0,
              orderTrackingId: trackingId,
              merchantReference: v.merchantReference || "",
              status: "successful",
              paymentMethod: v.paymentMethod || "",
              confirmationCode: v.confirmationCode || "",
              createdAt: v.createdAt || v.startDate || "",
              updatedAt: v.createdAt || "",
            }
          }
        }
      }

      setPayments(Object.values(paymentsMap))
      setLoadingData(false)
    }

    const unsubLogs = onValue(logsRef, (snap) => {
      logsData = snap.exists() ? snap.val() : null
      gotLogs = true
      buildPayments()
    }, () => { gotLogs = true; buildPayments() })

    const unsubSubs = onValue(subsRef, (snap) => {
      subsData = snap.exists() ? snap.val() : null
      gotSubs = true
      buildPayments()
    }, () => { gotSubs = true; buildPayments() })

    return () => {
      unsubLogs()
      unsubSubs()
    }
  }, [isAdmin])

  useEffect(() => {
    let result = [...payments]

    // Filter by status
    if (statusFilter === "successful") {
      result = result.filter((p) => p.status === "successful")
    } else if (statusFilter === "failed") {
      result = result.filter((p) => p.status === "failed" || p.status === "reversed")
    } else if (statusFilter === "pending") {
      result = result.filter((p) => p.status === "pending")
    }

    // Filter by search query (phone, email, name)
    if (searchQuery.trim()) {
      const q = searchQuery.toLowerCase()
      result = result.filter(
        (p) =>
          (p.phoneNumber || "").toLowerCase().includes(q) ||
          (p.email || "").toLowerCase().includes(q) ||
          (p.firstName || "").toLowerCase().includes(q) ||
          (p.lastName || "").toLowerCase().includes(q) ||
          (p.orderTrackingId || "").toLowerCase().includes(q)
      )
    }

    // Sort - always sort by date, newest first by default
    result.sort((a, b) => {
      // Handle missing or invalid dates - push them to the end
      const dateA = a.createdAt ? new Date(a.createdAt).getTime() : 0
      const dateB = b.createdAt ? new Date(b.createdAt).getTime() : 0
      
      // If both dates are invalid/missing, maintain original order
      if (dateA === 0 && dateB === 0) return 0
      // Push items without dates to the end
      if (dateA === 0) return 1
      if (dateB === 0) return -1
      
      return sortOrder === "newest" ? dateB - dateA : dateA - dateB
    })

    setFilteredPayments(result)
  }, [payments, statusFilter, searchQuery, sortOrder])

  const successfulPayments = payments.filter((p) => p.status === "successful")
  const failedPayments = payments.filter((p) => p.status === "failed" || p.status === "reversed")
  const pendingPayments = payments.filter((p) => p.status === "pending")
  const formatDate = (dateStr: string) => {
    if (!dateStr) return "N/A"
    const d = new Date(dateStr)
    return d.toLocaleDateString("en-UG", {
      year: "numeric",
      month: "short",
      day: "numeric",
    })
  }

  const formatTime = (dateStr: string) => {
    if (!dateStr) return ""
    const d = new Date(dateStr)
    return d.toLocaleTimeString("en-UG", {
      hour: "2-digit",
      minute: "2-digit",
      hour12: true,
    })
  }

  const getStatusBadge = (status: string) => {
    switch (status) {
      case "successful":
        return (
          <span className="inline-flex items-center gap-1 px-2 py-0.5 rounded-full text-[10px] font-semibold bg-emerald-500/15 text-emerald-400 border border-emerald-500/30">
            <CheckCircle2 className="w-3 h-3" />
            Successful
          </span>
        )
      case "failed":
      case "reversed":
        return (
          <span className="inline-flex items-center gap-1 px-2 py-0.5 rounded-full text-[10px] font-semibold bg-red-500/15 text-red-400 border border-red-500/30">
            <XCircle className="w-3 h-3" />
            {status === "reversed" ? "Reversed" : "Failed"}
          </span>
        )
      default:
        return (
          <span className="inline-flex items-center gap-1 px-2 py-0.5 rounded-full text-[10px] font-semibold bg-yellow-500/15 text-yellow-400 border border-yellow-500/30">
            <Clock className="w-3 h-3" />
            Pending
          </span>
        )
    }
  }

  const handleDeleteTransaction = async (payment: PaymentLog) => {
    if (!confirm(`Delete transaction for ${payment.firstName} ${payment.lastName} - UGX ${(payment.amount || 0).toLocaleString()}?`)) return
    try {
      const safeKey = payment.id.replace(/[.#$/\[\]]/g, "_")
      await remove(ref(database, `paymentLogs/${safeKey}`))
    } catch {
      alert("Failed to delete transaction")
    }
  }

  if (loading) {
    return (
      <div className="min-h-screen bg-background flex items-center justify-center">
        <div className="text-foreground text-xl">Loading...</div>
      </div>
    )
  }

  if (!isAdmin) return null

  return (
    <div className="min-h-screen bg-background">
      <AdminNav />
      <AdminMobileNav />

      <main className="pt-14 pb-20 md:pb-6 px-3 md:px-6 min-h-screen">
        <div className="max-w-7xl mx-auto">
          {/* Header */}
          <div className="mb-6">
            <div className="flex items-center gap-2 mb-1">
              <Wallet className="w-6 h-6 text-emerald-500" />
              <h1 className="text-2xl md:text-3xl font-bold text-foreground">Transactions</h1>
            </div>
            <p className="text-sm text-muted-foreground">All payment transactions with user details</p>
          </div>

          {/* Summary Cards */}
          <div className="grid grid-cols-3 gap-3 md:gap-4 mb-6">
            <Card className="bg-card border-border p-4">
              <div className="flex items-center gap-2 mb-2">
                <div className="w-8 h-8 rounded-full bg-emerald-500/15 flex items-center justify-center">
                  <CheckCircle2 className="w-4 h-4 text-emerald-500" />
                </div>
              </div>
              <div className="text-xl md:text-2xl font-bold text-emerald-400">{successfulPayments.length}</div>
              <div className="text-muted-foreground text-xs mt-1">Successful</div>
            </Card>

            <Card className="bg-card border-border p-4">
              <div className="flex items-center gap-2 mb-2">
                <div className="w-8 h-8 rounded-full bg-red-500/15 flex items-center justify-center">
                  <XCircle className="w-4 h-4 text-red-500" />
                </div>
              </div>
              <div className="text-xl md:text-2xl font-bold text-red-400">{failedPayments.length}</div>
              <div className="text-muted-foreground text-xs mt-1">Failed</div>
            </Card>

            <Card className="bg-card border-border p-4">
              <div className="flex items-center gap-2 mb-2">
                <div className="w-8 h-8 rounded-full bg-yellow-500/15 flex items-center justify-center">
                  <Clock className="w-4 h-4 text-yellow-500" />
                </div>
              </div>
              <div className="text-xl md:text-2xl font-bold text-yellow-400">{pendingPayments.length}</div>
              <div className="text-muted-foreground text-xs mt-1">Pending</div>
            </Card>
          </div>

          {/* Filters */}
          <Card className="bg-card border-border p-4 mb-4">
            <div className="flex flex-col sm:flex-row gap-3">
              {/* Search */}
              <div className="flex-1 relative">
                <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-muted-foreground" />
                <input
                  type="text"
                  placeholder="Search by phone, email, or name..."
                  value={searchQuery}
                  onChange={(e) => setSearchQuery(e.target.value)}
                  className="w-full pl-9 pr-4 py-2 bg-muted/50 border border-border rounded-lg text-sm text-foreground placeholder:text-muted-foreground focus:outline-none focus:ring-2 focus:ring-emerald-500/50"
                />
              </div>

              {/* Status Filter */}
              <div className="flex items-center gap-2">
                <Filter className="w-4 h-4 text-muted-foreground" />
                <select
                  value={statusFilter}
                  onChange={(e) => setStatusFilter(e.target.value as any)}
                  className="bg-muted/50 border border-border rounded-lg px-3 py-2 text-sm text-foreground focus:outline-none focus:ring-2 focus:ring-emerald-500/50"
                >
                  <option value="all">All Status</option>
                  <option value="successful">Successful</option>
                  <option value="failed">Failed</option>
                  <option value="pending">Pending</option>
                </select>
              </div>

              {/* Sort */}
              <button
                onClick={() => setSortOrder(sortOrder === "newest" ? "oldest" : "newest")}
                className="flex items-center gap-1 px-3 py-2 bg-muted/50 border border-border rounded-lg text-sm text-foreground hover:bg-muted transition"
              >
                <ArrowUpDown className="w-4 h-4" />
                {sortOrder === "newest" ? "Newest" : "Oldest"}
              </button>
            </div>
          </Card>

          {/* Payments List */}
          <Card className="bg-card border-border overflow-hidden">
            {loadingData ? (
              <div className="p-8 text-center">
                <div className="w-8 h-8 border-4 border-muted-foreground border-t-foreground rounded-full animate-spin mx-auto mb-3"></div>
                <p className="text-muted-foreground text-sm">Loading payments...</p>
              </div>
            ) : filteredPayments.length === 0 ? (
              <div className="p-8 text-center">
                <Wallet className="w-12 h-12 text-muted-foreground mx-auto mb-3 opacity-50" />
                <p className="text-muted-foreground text-sm">No payments found</p>
              </div>
            ) : (
              <>
                {/* Desktop Table */}
                <div className="hidden md:block overflow-x-auto">
                  <table className="w-full">
                    <thead>
                      <tr className="border-b border-border bg-muted/30">
                        <th className="text-left px-4 py-3 text-xs font-semibold text-muted-foreground uppercase">User</th>
                        <th className="text-left px-4 py-3 text-xs font-semibold text-muted-foreground uppercase">Contact</th>
                        <th className="text-left px-4 py-3 text-xs font-semibold text-muted-foreground uppercase">Plan</th>
                        <th className="text-left px-4 py-3 text-xs font-semibold text-muted-foreground uppercase">Amount</th>
                        <th className="text-left px-4 py-3 text-xs font-semibold text-muted-foreground uppercase">Method</th>
                        <th className="text-left px-4 py-3 text-xs font-semibold text-muted-foreground uppercase">Status</th>
                        <th className="text-left px-4 py-3 text-xs font-semibold text-muted-foreground uppercase">Date & Time</th>
                        <th className="text-right px-4 py-3 text-xs font-semibold text-muted-foreground uppercase">Action</th>
                      </tr>
                    </thead>
                    <tbody className="divide-y divide-border">
                      {filteredPayments.map((payment) => (
                        <tr key={payment.id} className="hover:bg-muted/20 transition-colors">
                          <td className="px-4 py-3">
                            <div className="flex items-center gap-2">
                              <div className="w-8 h-8 rounded-full bg-muted flex items-center justify-center">
                                <User className="w-4 h-4 text-muted-foreground" />
                              </div>
                              <div>
                                <p className="text-sm font-medium text-foreground">
                                  {payment.firstName || ""} {payment.lastName || ""}
                                </p>
                                <p className="text-[10px] text-muted-foreground truncate max-w-[120px]">
                                  {payment.orderTrackingId}
                                </p>
                              </div>
                            </div>
                          </td>
                          <td className="px-4 py-3">
                            <div className="space-y-0.5">
                              {payment.phoneNumber && (
                                <div className="flex items-center gap-1 text-sm text-foreground">
                                  <Phone className="w-3 h-3 text-muted-foreground" />
                                  {payment.phoneNumber}
                                </div>
                              )}
                              {payment.email && (
                                <div className="flex items-center gap-1 text-xs text-muted-foreground">
                                  <Mail className="w-3 h-3" />
                                  <span className="truncate max-w-[150px]">{payment.email}</span>
                                </div>
                              )}
                            </div>
                          </td>
                          <td className="px-4 py-3">
                            <span className="text-sm text-foreground">{payment.planName || "N/A"}</span>
                            <p className="text-[10px] text-muted-foreground">{payment.days} day{payment.days !== 1 ? "s" : ""}</p>
                          </td>
                          <td className="px-4 py-3">
                            <span className="text-sm font-semibold text-foreground">
                              UGX {(payment.amount || 0).toLocaleString()}
                            </span>
                          </td>
                          <td className="px-4 py-3">
                            <span className="text-xs text-foreground">{payment.paymentMethod || "N/A"}</span>
                            {payment.confirmationCode && (
                              <p className="text-[10px] text-muted-foreground font-mono">{payment.confirmationCode}</p>
                            )}
                          </td>
                          <td className="px-4 py-3">{getStatusBadge(payment.status)}</td>
                          <td className="px-4 py-3">
                            <p className="text-sm text-foreground">{formatDate(payment.createdAt)}</p>
                            <p className="text-[10px] text-muted-foreground">{formatTime(payment.createdAt)}</p>
                          </td>
                          <td className="px-4 py-3 text-right">
                            <button
                              onClick={() => handleDeleteTransaction(payment)}
                              className="p-1.5 rounded-md text-muted-foreground hover:text-red-400 hover:bg-red-500/10 transition"
                              title="Delete transaction"
                            >
                              <Trash2 className="w-4 h-4" />
                            </button>
                          </td>
                        </tr>
                      ))}
                    </tbody>
                  </table>
                </div>

                {/* Mobile Cards */}
                <div className="md:hidden divide-y divide-border">
                  {filteredPayments.map((payment) => (
                    <div key={payment.id} className="p-3">
                      <div className="flex items-start justify-between mb-2">
                        <div className="flex items-center gap-2">
                          <div className="w-8 h-8 rounded-full bg-muted flex items-center justify-center">
                            <User className="w-4 h-4 text-muted-foreground" />
                          </div>
                          <div>
                            <p className="text-sm font-medium text-foreground">
                              {payment.firstName || ""} {payment.lastName || ""}
                            </p>
                            <p className="text-[10px] text-muted-foreground">{formatDate(payment.createdAt)} at {formatTime(payment.createdAt)}</p>
                          </div>
                        </div>
                        {getStatusBadge(payment.status)}
                      </div>

                      <div className="flex flex-wrap items-center gap-x-4 gap-y-1 text-xs text-muted-foreground ml-10">
                        {payment.phoneNumber && (
                          <span className="flex items-center gap-1">
                            <Phone className="w-3 h-3" />
                            {payment.phoneNumber}
                          </span>
                        )}
                        {payment.email && (
                          <span className="flex items-center gap-1">
                            <Mail className="w-3 h-3" />
                            <span className="truncate max-w-[150px]">{payment.email}</span>
                          </span>
                        )}
                      </div>

                      <div className="flex items-center justify-between mt-2 ml-10">
                        <div>
                          <span className="text-xs text-muted-foreground">{payment.planName || "N/A"} ({payment.days}d)</span>
                          {payment.paymentMethod && (
                            <span className="text-[10px] text-muted-foreground ml-2">via {payment.paymentMethod}</span>
                          )}
                        </div>
                        <span className="text-sm font-bold text-foreground">UGX {(payment.amount || 0).toLocaleString()}</span>
                      </div>
                      {payment.confirmationCode && (
                        <p className="text-[10px] text-muted-foreground ml-10 mt-0.5 font-mono">Ref: {payment.confirmationCode}</p>
                      )}
                      <div className="flex justify-end mt-2">
                        <button
                          onClick={() => handleDeleteTransaction(payment)}
                          className="flex items-center gap-1 px-2 py-1 rounded text-[10px] text-red-400 hover:bg-red-500/10 transition"
                        >
                          <Trash2 className="w-3 h-3" />
                          Delete
                        </button>
                      </div>
                    </div>
                  ))}
                </div>
              </>
            )}
          </Card>

          <p className="text-center text-muted-foreground text-xs mt-4">
            Showing {filteredPayments.length} of {payments.length} payments
          </p>
        </div>
      </main>
    </div>
  )
}
