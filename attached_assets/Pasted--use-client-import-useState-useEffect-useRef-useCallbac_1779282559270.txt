"use client"

import { useState, useEffect, useRef, useCallback } from "react"
import { useRouter } from "next/navigation"
import { useAuth } from "@/lib/auth-context"
import { useSubscription, SUBSCRIPTION_PLANS } from "@/lib/subscription-context"
import { database } from "@/lib/firebase"
import { ref, set, update } from "firebase/database"
import { Check, X, Loader2, CheckCircle, XCircle, ArrowLeft, ShieldCheck } from "lucide-react"

type ModalPhase = "plans" | "phone" | "paying" | "success" | "failed"

interface SubscriptionModalProps {
  isOpen: boolean
  onClose: () => void
}

export default function SubscriptionModal({ isOpen, onClose }: SubscriptionModalProps) {
  const router = useRouter()
  const { user } = useAuth()
  const { refreshSubscription } = useSubscription()
  const [processingPlan, setProcessingPlan] = useState<string | null>(null)
  const [error, setError] = useState("")

  const [phase, setPhase] = useState<ModalPhase>("plans")
  const [iframeUrl, setIframeUrl] = useState("")
  const [pendingPlan, setPendingPlan] = useState<any>(null)
  const [statusMessage, setStatusMessage] = useState("")
  const [pollCount, setPollCount] = useState(0)
  const [phoneNumber, setPhoneNumber] = useState("")
  const [selectedPlanForPhone, setSelectedPlanForPhone] = useState<any>(null)
  const pollRef = useRef<ReturnType<typeof setInterval> | null>(null)
  const activatedRef = useRef(false)

  useEffect(() => {
    if (!isOpen) {
      // Don't kill polling - the global poller in subscription-context
      // will continue checking silently and auto-activate when successful
      if (phase !== "success") {
        setPhase("plans")
        setIframeUrl("")
        setPendingPlan(null)
        setError("")
        setProcessingPlan(null)
        activatedRef.current = false
        setPollCount(0)
      }
    }
  }, [isOpen, phase])

  useEffect(() => {
    return () => {
      if (pollRef.current) clearInterval(pollRef.current)
    }
  }, [])

  const activateSubscription = useCallback(async (
    planData: any,
    trackingId: string,
    paymentMethod: string,
    confirmationCode: string,
    paymentAccount?: string,
    amount?: number
  ) => {
    if (activatedRef.current || !user) return
    activatedRef.current = true

    // Show success immediately
    setPhase("success")
    setStatusMessage("Payment successful! Enjoy watching.")
    localStorage.removeItem("pendingSubscription")

    // Auto-close after 2 seconds
    setTimeout(() => {
      onClose()
      setPhase("plans")
    }, 2000)

    try {
      const startDate = new Date()
      const endDate = new Date()
      endDate.setDate(endDate.getDate() + planData.days)

      // Use the user-entered phone number from planData, not from PesaPal
      const userPhone = planData.phoneNumber || paymentAccount || ""

      const subscriptionData = {
        planId: planData.planId || planData.id,
        planName: planData.planName || planData.name,
        amount: amount || planData.amount || planData.price,
        days: planData.days,
        orderTrackingId: trackingId,
        merchantReference: planData.merchantReference || "",
        confirmationCode,
        paymentMethod,
        phoneNumber: userPhone,
        startDate: startDate.toISOString(),
        endDate: endDate.toISOString(),
        active: true,
        createdAt: new Date().toISOString(),
      }

      const safeKey = trackingId.replace(/[.#$/\[\]]/g, "_")
      await Promise.all([
        set(ref(database, `subscriptions/${user.uid}`), subscriptionData),
        update(ref(database, `paymentLogs/${safeKey}`), {
          phoneNumber: userPhone,
          amount: amount || planData.amount || planData.price,
          status: "successful",
          paymentMethod,
          confirmationCode,
          updatedAt: new Date().toISOString(),
        }),
      ])

      refreshSubscription()
    } catch {
      refreshSubscription()
    }
  }, [user, refreshSubscription, onClose])

  const [lastStatusDesc, setLastStatusDesc] = useState("")
  const [callbackDetected, setCallbackDetected] = useState(false)

  // Listen for postMessage from the callback page loaded inside the iframe
  useEffect(() => {
    const handleMessage = (event: MessageEvent) => {
      if (event.data && event.data.type === "pesapal-callback") {
        setCallbackDetected(true)
        setStatusMessage("Payment submitted! Verifying with PesaPal...")
      }
    }
    window.addEventListener("message", handleMessage)
    return () => window.removeEventListener("message", handleMessage)
  }, [])

  const startPolling = useCallback((trackingId: string, planData: any) => {
    if (pollRef.current) clearInterval(pollRef.current)
    let attempts = 0
    let polling = true
    let fetching = false // prevent overlapping fetches

    const checkStatus = async () => {
      if (!polling || activatedRef.current || fetching) return
      fetching = true

      attempts++
      setPollCount(attempts)

      try {
        const res = await fetch(`/api/pesapal/transaction-status?orderTrackingId=${trackingId}&_t=${Date.now()}`, {
          cache: "no-store",
        })
        const data = await res.json()
        const statusCode = data.status_code ?? data.statusCode ?? -1

        // Map "INVALID" to user-friendly text
        let friendlyStatus = data.payment_status_description || ""
        if (statusCode === 0) {
          friendlyStatus = "Pending Payment"
        } else if (statusCode === 1) {
          friendlyStatus = "Completed"
        } else if (statusCode === 2) {
          friendlyStatus = "Failed"
        } else if (statusCode === 3) {
          friendlyStatus = "Reversed"
        }
        setLastStatusDesc(friendlyStatus)

        if (statusCode === 1) {
          polling = false
          if (pollRef.current) clearInterval(pollRef.current)
          activateSubscription(
            planData, trackingId,
            data.payment_method || "PesaPal",
            data.confirmation_code || "",
            data.payment_account || "",
            data.amount || 0
          )
        } else if (statusCode === 2 || statusCode === 3) {
          polling = false
          if (pollRef.current) clearInterval(pollRef.current)
          setPhase("failed")
          setStatusMessage(statusCode === 3 ? "Payment was reversed." : (data.description || "Payment failed. Please try again."))

          // Update payment log with failure status for admin visibility
          if (user) {
            const safeKey = trackingId.replace(/[.#$/\[\]]/g, "_")
            update(ref(database, `paymentLogs/${safeKey}`), {
              phoneNumber: data.payment_account || "",
              amount: data.amount || planData.amount || planData.price || 0,
              status: statusCode === 3 ? "reversed" : "failed",
              paymentMethod: data.payment_method || "",
              confirmationCode: data.confirmation_code || "",
              updatedAt: new Date().toISOString(),
            }).catch(() => {})
          }

          localStorage.removeItem("pendingSubscription")
        }
      } catch {
        // Network error, keep trying
      } finally {
        fetching = false
      }
    }

    pollRef.current = setInterval(checkStatus, 1500)
    checkStatus()
  }, [activateSubscription])

  const handleSelectPlan = (planId: string) => {
    if (!user) {
      onClose()
      router.push("/login")
      return
    }

    const plan = SUBSCRIPTION_PLANS.find((p) => p.id === planId)
    if (!plan) {
      setError("Invalid plan selected")
      return
    }

    setSelectedPlanForPhone(plan)
    setPhoneNumber("")
    setError("")
    setPhase("phone")
  }

  const handlePhoneSubmit = async (e: React.FormEvent) => {
    e.preventDefault()

    if (!phoneNumber.trim()) {
      setError("Please enter your phone number")
      return
    }

    if (!user || !selectedPlanForPhone) return

    setProcessingPlan(selectedPlanForPhone.id)
    setError("")
    activatedRef.current = false
    setPollCount(0)

    try {
      const baseUrl = window.location.origin

      const response = await fetch("/api/pesapal/submit-order", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          amount: selectedPlanForPhone.price,
          phoneNumber: phoneNumber.trim(),
          email: user.email || "",
          firstName: user.displayName?.split(" ")[0] || "Customer",
          lastName: user.displayName?.split(" ").slice(1).join(" ") || "",
          description: `VJ PILES UG - ${selectedPlanForPhone.name} Subscription`,
          callbackUrl: `${baseUrl}/subscribe/callback`,
          cancellationUrl: `${baseUrl}/subscribe`,
          userId: user.uid,
          planId: selectedPlanForPhone.id,
          planName: selectedPlanForPhone.name,
          days: selectedPlanForPhone.days,
        }),
      })

      const data = await response.json()
      if (!data.success) throw new Error(data.error || "Failed to create payment order")

      const planData = {
        planId: selectedPlanForPhone.id,
        planName: selectedPlanForPhone.name,
        amount: selectedPlanForPhone.price,
        days: selectedPlanForPhone.days,
        email: user.email || "",
        phoneNumber: phoneNumber.trim(),
        orderTrackingId: data.orderTrackingId,
        merchantReference: data.merchantReference,
      }

      // Store plan data locally
      localStorage.setItem("pendingSubscription", JSON.stringify(planData))

      // Save to paymentLogs immediately with status "pending" so admin sees it right away
      const safeKey = data.orderTrackingId.replace(/[.#$/\[\]]/g, "_")
      set(ref(database, `paymentLogs/${safeKey}`), {
        userId: user.uid,
        email: user.email || "",
        firstName: user.displayName?.split(" ")[0] || "Customer",
        lastName: user.displayName?.split(" ").slice(1).join(" ") || "",
        phoneNumber: phoneNumber.trim(),
        amount: selectedPlanForPhone.price,
        planId: selectedPlanForPhone.id,
        planName: selectedPlanForPhone.name,
        days: selectedPlanForPhone.days,
        orderTrackingId: data.orderTrackingId,
        merchantReference: data.merchantReference || "",
        status: "pending",
        paymentMethod: "",
        confirmationCode: "",
        createdAt: new Date().toISOString(),
      }).catch(() => {})

      setPendingPlan(planData)
      setIframeUrl(data.redirectUrl)
      setPhase("paying")
      setProcessingPlan(null)
      startPolling(data.orderTrackingId, planData)
    } catch (err: any) {
      setError(err.message || "Payment failed. Please try again.")
      setProcessingPlan(null)
    }
  }

  const handleBack = () => {
    // Don't clear polling - global poller continues silently
    if (phase === "phone") {
      setPhase("plans")
      setSelectedPlanForPhone(null)
      setPhoneNumber("")
    } else {
      setPhase("plans")
      setIframeUrl("")
    }
    activatedRef.current = false
    setPollCount(0)
  }

  if (!isOpen) return null

  return (
    <div
      className="fixed inset-0 bg-black/70 backdrop-blur-sm z-50 flex items-end sm:items-center justify-center"
      onClick={phase === "plans" ? onClose : undefined}
    >
      <div
        className={`bg-card border border-border rounded-t-2xl sm:rounded-2xl shadow-2xl relative animate-in slide-in-from-bottom-4 fade-in duration-200 overflow-hidden ${
          phase === "paying"
            ? "w-full max-w-[380px] sm:max-w-[400px] flex flex-col"
            : phase === "phone"
            ? "w-full max-w-[320px] sm:max-w-[360px] max-h-[90vh] overflow-y-auto"
            : "w-full max-w-2xl max-h-[90vh] overflow-y-auto"
        }`}
        style={phase === "paying" ? { height: "min(92vh, 680px)" } : undefined}
        onClick={(e) => e.stopPropagation()}
      >
        {/* Success */}
        {phase === "success" && (
          <div className="p-8 text-center">
            <div className="w-16 h-16 bg-green-500/10 rounded-full flex items-center justify-center mx-auto mb-4">
              <CheckCircle className="w-8 h-8 text-green-500" />
            </div>
            <h2 className="text-xl font-bold text-foreground mb-2">Payment Successful!</h2>
            <p className="text-muted-foreground text-sm">{statusMessage}</p>
          </div>
        )}

        {/* Failed */}
        {phase === "failed" && (
          <div className="p-8 text-center">
            <div className="w-16 h-16 bg-red-500/10 rounded-full flex items-center justify-center mx-auto mb-4">
              <XCircle className="w-8 h-8 text-red-500" />
            </div>
            <h2 className="text-xl font-bold text-foreground mb-2">Payment Issue</h2>
            <p className="text-muted-foreground text-sm mb-5">{statusMessage}</p>
            <button
              onClick={handleBack}
              className="px-6 py-2.5 bg-primary hover:bg-primary/90 text-primary-foreground rounded-lg font-semibold transition text-sm"
            >
              Try Again
            </button>
          </div>
        )}

        {/* Phone Input */}
        {phase === "phone" && (
          <div className="px-3 sm:px-4 py-2 sm:py-3">
            <div className="flex items-center justify-between mb-2 sm:mb-3">
              <h2 className="text-xs sm:text-sm font-semibold text-foreground">Phone Number</h2>
              <button
                onClick={handleBack}
                className="text-muted-foreground hover:text-foreground transition text-xs"
              >
                ✕
              </button>
            </div>

            <form onSubmit={handlePhoneSubmit} className="space-y-2">
              <div className="max-w-xs">
                <input
                  type="tel"
                  placeholder="07xxxxxxxxx"
                  value={phoneNumber}
                  onChange={(e) => setPhoneNumber(e.target.value)}
                  className="w-full px-2.5 sm:px-3 py-1.5 sm:py-2 bg-muted/50 border border-border rounded-md text-xs sm:text-sm text-foreground placeholder:text-muted-foreground focus:outline-none focus:ring-2 focus:ring-primary/50 transition"
                />
                <p className="text-xs text-muted-foreground mt-0.5">E.g. 0700123456 or 256700123456</p>
              </div>

              {error && (
                <div className="p-1.5 sm:p-2 bg-red-500/10 border border-red-500/20 rounded-md">
                  <p className="text-xs text-red-400">{error}</p>
                </div>
              )}

              <div className="bg-muted/20 rounded-md p-2 space-y-0.5 max-w-xs">
                <div className="flex items-center justify-between gap-2">
                  <span className="text-xs font-medium text-foreground truncate">{selectedPlanForPhone?.name}</span>
                  <span className="text-xs sm:text-sm font-bold text-primary whitespace-nowrap">UGX {selectedPlanForPhone?.price?.toLocaleString()}</span>
                </div>
                <p className="text-xs text-muted-foreground">{selectedPlanForPhone?.days}d pass</p>
              </div>

              <button
                type="submit"
                disabled={!phoneNumber.trim() || processingPlan === selectedPlanForPhone?.id}
                className="max-w-xs px-3 py-1.5 sm:py-2 bg-primary hover:bg-primary/90 disabled:opacity-50 disabled:cursor-not-allowed text-primary-foreground rounded-md font-medium transition flex items-center justify-center gap-1 text-xs sm:text-sm"
              >
                {processingPlan === selectedPlanForPhone?.id ? (
                  <>
                    <Loader2 className="w-3 h-3 animate-spin" />
                    <span>Processing...</span>
                  </>
                ) : (
                  <span>Proceed</span>
                )}
              </button>
            </form>
          </div>
        )}

        {/* Paying - embedded iframe */}
        {phase === "paying" && (
          <>
            <div className="flex items-center justify-between px-3 py-1.5 border-b border-border bg-card/80 flex-shrink-0">
              <button
                onClick={handleBack}
                className="flex items-center gap-1 text-muted-foreground hover:text-foreground transition text-[11px]"
              >
                <ArrowLeft className="w-3 h-3" />
                <span>Back</span>
              </button>
              <div className="flex items-center gap-1">
                <ShieldCheck className="w-3 h-3 text-green-500" />
                <span className="text-[10px] font-medium text-foreground">Secure Payment</span>
              </div>
              <button
                onClick={onClose}
                className="w-5 h-5 rounded-full bg-muted hover:bg-accent flex items-center justify-center transition"
              >
                <X className="w-2.5 h-2.5 text-foreground" />
              </button>
            </div>

            <div className="flex-1 relative min-h-0 overflow-hidden">
              <iframe
                src={iframeUrl}
                style={{
                  position: "absolute",
                  top: 0,
                  left: 0,
                  width: "125%",
                  height: "125%",
                  transform: "scale(0.8)",
                  transformOrigin: "top left",
                  border: "none",
                }}
                allow="payment"
                sandbox="allow-scripts allow-same-origin allow-forms allow-popups allow-top-navigation"
                title="PesaPal Payment"
              />
            </div>

            <div className="px-3 py-1.5 border-t border-border bg-card flex-shrink-0">
              <div className="flex items-center gap-2">
                <Loader2 className="w-3 h-3 text-primary animate-spin flex-shrink-0" />
                <div className="flex-1 min-w-0">
                  <p className="text-[11px] font-medium text-foreground truncate">
                    {statusMessage || (callbackDetected ? "Verifying payment..." : "Waiting for payment...")}
                  </p>
                  <p className="text-[10px] text-muted-foreground truncate">
                    {callbackDetected
                      ? `Confirming... (check #${pollCount})`
                      : lastStatusDesc
                        ? `${lastStatusDesc} (check #${pollCount})`
                        : `Complete payment above (check #${pollCount})`}
                  </p>
                </div>
                {pendingPlan && (
                  <div className="flex-shrink-0 text-right">
                    <p className="text-[10px] text-muted-foreground">{pendingPlan.planName}</p>
                    <p className="text-[11px] font-bold text-foreground">UGX {pendingPlan.amount?.toLocaleString()}</p>
                  </div>
                )}
              </div>
            </div>
          </>
        )}

        {/* Plan selection */}
        {phase === "plans" && (
          <div className="p-5">
            <button
              onClick={onClose}
              className="absolute top-3 right-3 w-7 h-7 rounded-full bg-muted hover:bg-accent flex items-center justify-center transition z-10"
            >
              <X className="w-4 h-4 text-foreground" />
            </button>

            <div className="text-center mb-4">
              <img
                src="https://i.postimg.cc/sgHQjqdW/Vj-Piles-U3333.png"
                alt="VJ PILES UG"
                className="w-20 h-20 rounded-2xl object-cover mx-auto mb-2"
              />
              <h2 className="text-xl font-bold text-foreground mb-1">Subscribe to VJ PILES UG</h2>
              <p className="text-muted-foreground text-xs">Tap a plan to pay via PesaPal</p>
            </div>

            {error && (
              <div className="mb-4 bg-destructive/10 border border-destructive/30 rounded-lg p-3">
                <p className="text-destructive text-sm text-center">{error}</p>
              </div>
            )}

            <div className="grid grid-cols-2 sm:grid-cols-3 lg:grid-cols-5 gap-2 mb-4">
              {SUBSCRIPTION_PLANS.map((plan) => {
                const isPopular = plan.id === "1week"
                const isProcessing = processingPlan === plan.id

                return (
                  <button
                    key={plan.id}
                    onClick={() => handleSelectPlan(plan.id)}
                    disabled={processingPlan !== null}
                    className={`relative p-3 rounded-lg border-2 transition ${
                      isProcessing
                        ? "border-primary bg-primary/5"
                        : "border-border bg-card hover:border-primary"
                    } ${processingPlan !== null && !isProcessing ? "opacity-50 cursor-not-allowed" : ""}`}
                  >
                    {isPopular && (
                      <div className="absolute -top-1 left-1/2 -translate-x-1/2 px-1.5 py-0.5 bg-primary text-primary-foreground text-[10px] font-bold rounded">
                        Popular
                      </div>
                    )}

                    {isProcessing && (
                      <div className="absolute inset-0 flex items-center justify-center bg-background/80 rounded-lg z-10">
                        <Loader2 className="w-5 h-5 text-primary animate-spin" />
                      </div>
                    )}

                    <h3 className="text-xs font-bold text-foreground mb-1">{plan.name}</h3>
                    <div className="flex items-baseline justify-center gap-0.5 mb-2">
                      <span className="text-[10px] text-muted-foreground">UGX</span>
                      <span className="text-sm font-bold text-foreground">{plan.price.toLocaleString()}</span>
                    </div>

                    <div className="space-y-0.5 text-[10px]">
                      <div className="flex items-center gap-1 text-muted-foreground">
                        <Check className="w-2.5 h-2.5 text-primary" />
                        <span>Unlimited</span>
                      </div>
                      <div className="flex items-center gap-1 text-muted-foreground">
                        <Check className="w-2.5 h-2.5 text-primary" />
                        <span>Downloads</span>
                      </div>
                    </div>
                  </button>
                )
              })}
            </div>

            <p className="text-muted-foreground text-[10px] text-center">
              Secure payment via <span className="text-foreground font-semibold">PesaPal</span> - Mobile Money, Card, or Bank
            </p>
          </div>
        )}
      </div>
    </div>
  )
}
