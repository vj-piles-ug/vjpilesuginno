import { NextResponse } from "next/server"

const PESAPAL_CONSUMER_KEY = process.env.PESAPAL_CONSUMER_KEY || "vTibxQeLvkiJ0aBZQk3Ci5gL2BUx1dP0"
const PESAPAL_CONSUMER_SECRET = process.env.PESAPAL_CONSUMER_SECRET || "kdwghPqFeT5dxwnhrpVccfY9B7o="
const PESAPAL_API_URL = "https://pay.pesapal.com/v3"

// Cache the auth token so we don't request a new one on every poll
let cachedToken: string | null = null
let tokenExpiry: number = 0

async function getAuthToken(): Promise<string | null> {
  // Return cached token if still valid (with 60s buffer)
  if (cachedToken && Date.now() < tokenExpiry - 60000) {
    return cachedToken
  }

  try {
    const response = await fetch(`${PESAPAL_API_URL}/api/Auth/RequestToken`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        Accept: "application/json",
      },
      body: JSON.stringify({
        consumer_key: PESAPAL_CONSUMER_KEY,
        consumer_secret: PESAPAL_CONSUMER_SECRET,
      }),
    })

    const data = await response.json()
    if (data.token) {
      cachedToken = data.token
      // PesaPal tokens typically expire in 5 minutes; parse expiryDate or default to 4 min
      if (data.expiryDate) {
        tokenExpiry = new Date(data.expiryDate).getTime()
      } else {
        tokenExpiry = Date.now() + 4 * 60 * 1000
      }
      return cachedToken
    }
    return null
  } catch {
    return null
  }
}

export async function GET(request: Request) {
  try {
    const { searchParams } = new URL(request.url)
    const orderTrackingId = searchParams.get("orderTrackingId")

    if (!orderTrackingId) {
      return NextResponse.json(
        { success: false, error: "Order tracking ID is required", status_code: -1 },
        { status: 400 }
      )
    }

    const token = await getAuthToken()
    if (!token) {
      return NextResponse.json(
        { success: false, error: "Auth failed", status_code: -1 },
        { status: 401 }
      )
    }

    let response: Response
    try {
      response = await fetch(
        `${PESAPAL_API_URL}/api/Transactions/GetTransactionStatus?orderTrackingId=${orderTrackingId}`,
        {
          method: "GET",
          cache: "no-store",
          headers: {
            "Content-Type": "application/json",
            Accept: "application/json",
            Authorization: `Bearer ${token}`,
          },
        }
      )
    } catch (fetchErr: any) {
      // Network-level failure calling PesaPal - return clean pending response
      return NextResponse.json(
        { success: false, error: fetchErr.message || "Network error", status_code: -1 },
        { status: 200, headers: { "Cache-Control": "no-store, no-cache, must-revalidate, max-age=0" } }
      )
    }

    if (!response.ok) {
      // If 401, token may have expired - clear cache so next call gets new token
      if (response.status === 401) {
        cachedToken = null
        tokenExpiry = 0
      }
      return NextResponse.json(
        { success: false, error: `PesaPal returned ${response.status}`, status_code: -1 },
        { status: 200, headers: { "Cache-Control": "no-store, no-cache, must-revalidate, max-age=0" } }
      )
    }

    const data = await response.json()

    // PesaPal status_code: 0=INVALID, 1=COMPLETED, 2=FAILED, 3=REVERSED
    let statusCode = -1
    if (typeof data.status_code === "number") {
      statusCode = data.status_code
    } else if (typeof data.payment_status_code === "number") {
      statusCode = data.payment_status_code
    } else if (typeof data.status_code === "string" && data.status_code !== "") {
      statusCode = parseInt(data.status_code, 10)
      if (isNaN(statusCode)) statusCode = -1
    }

    // Fallback: check description text
    if (statusCode === -1 && data.payment_status_description) {
      const desc = data.payment_status_description.toLowerCase()
      if (desc === "completed") statusCode = 1
      else if (desc === "failed") statusCode = 2
      else if (desc === "reversed") statusCode = 3
      else if (desc === "invalid") statusCode = 0
    }

    return NextResponse.json({
      success: true,
      status_code: statusCode,
      statusCode: statusCode,
      payment_status_description: data.payment_status_description || "",
      description: data.description || "",
      merchant_reference: data.merchant_reference || "",
      payment_method: data.payment_method || "",
      amount: data.amount || 0,
      currency: data.currency || "",
      confirmation_code: data.confirmation_code || "",
      payment_account: data.payment_account || "",
      message: data.message || "",
    }, {
      headers: {
        "Cache-Control": "no-store, no-cache, must-revalidate, max-age=0",
        "Pragma": "no-cache",
      }
    })
  } catch (error: any) {
    console.error("[v0] transaction-status error:", error.message)
    return NextResponse.json(
      { success: false, error: error.message, status_code: -1 },
      { status: 500 }
    )
  }
}
