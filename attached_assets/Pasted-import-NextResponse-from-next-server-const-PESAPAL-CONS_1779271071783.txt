import { NextResponse } from "next/server"

const PESAPAL_CONSUMER_KEY = process.env.PESAPAL_CONSUMER_KEY || "vTibxQeLvkiJ0aBZQk3Ci5gL2BUx1dP0"
const PESAPAL_CONSUMER_SECRET = process.env.PESAPAL_CONSUMER_SECRET || "kdwghPqFeT5dxwnhrpVccfY9B7o="
const PESAPAL_API_URL = "https://pay.pesapal.com/v3"

async function getAuthToken() {
  const response = await fetch(`${PESAPAL_API_URL}/api/Auth/RequestToken`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      Accept: "application/json",
    },
    body: JSON.stringify({
      consumer_key: PESAPAL_CONSUMER_KEY,
      consumer_secret: PESAPAL_CONSUMER_SECRET,
    }),
  })
  const data = await response.json()
  return data.token
}

export async function POST(request: Request) {
  try {
    const body = await request.json()
    const { 
      amount, 
      phoneNumber, 
      description, 
      email, 
      firstName, 
      lastName,
      callbackUrl,
      cancellationUrl,
      userId,
      planId,
      planName,
      days
    } = body

    const token = await getAuthToken()
    if (!token) {
      return NextResponse.json({ success: false, error: "Failed to authenticate with PesaPal" }, { status: 401 })
    }

    // Register IPN URL - always point to the server-side IPN handler
    const appUrl = process.env.NEXT_PUBLIC_APP_URL || 'https://vjpilesug.vercel.app'
    const ipnUrl = `${appUrl}/api/pesapal/ipn`

    const ipnResponse = await fetch(`${PESAPAL_API_URL}/api/URLSetup/RegisterIPN`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        Accept: "application/json",
        Authorization: `Bearer ${token}`,
      },
      body: JSON.stringify({
        url: ipnUrl,
        ipn_notification_type: "GET",
      }),
    })

    const ipnData = await ipnResponse.json()
    const ipnId = ipnData.ipn_id

    if (!ipnId) {
      console.error("Failed to register IPN:", ipnData)
      return NextResponse.json({ success: false, error: "Failed to setup payment notification" }, { status: 400 })
    }

    // Generate unique merchant reference
    const merchantReference = `VJPILES-${Date.now()}-${Math.random().toString(36).substring(7)}`

    // Submit order request
    // redirect_mode PARENT_WINDOW keeps redirect inside the iframe
    const orderRequest = {
      id: merchantReference,
      currency: "UGX",
      amount: amount,
      description: description || "VJ PILES UG Subscription",
      callback_url: callbackUrl || `${appUrl}/subscribe/callback`,
      cancellation_url: cancellationUrl || `${appUrl}/subscribe`,
      redirect_mode: "PARENT_WINDOW",
      notification_id: ipnId,
      billing_address: {
        email_address: email || "",
        phone_number: phoneNumber || "",
        country_code: "UG",
        first_name: firstName || "Customer",
        last_name: lastName || "",
        line_1: "",
        line_2: "",
        city: "",
        state: "",
        postal_code: "",
        zip_code: "",
      },
    }

    const orderResponse = await fetch(`${PESAPAL_API_URL}/api/Transactions/SubmitOrderRequest`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        Accept: "application/json",
        Authorization: `Bearer ${token}`,
      },
      body: JSON.stringify(orderRequest),
    })

    const orderData = await orderResponse.json()

    if (orderData.order_tracking_id && orderData.redirect_url) {
      // Return order info - client will save pending order to Firebase using client SDK
      // (Server-side REST writes to Firebase get 401 in this runtime)
      return NextResponse.json({
        success: true,
        orderTrackingId: orderData.order_tracking_id,
        merchantReference: orderData.merchant_reference || merchantReference,
        redirectUrl: orderData.redirect_url,
      })
    } else {
      console.error("PesaPal Order Error:", orderData)
      return NextResponse.json({ 
        success: false, 
        error: orderData.error?.message || orderData.message || "Failed to create payment order" 
      }, { status: 400 })
    }
  } catch (error: any) {
    console.error("PesaPal Submit Order Error:", error)
    return NextResponse.json({ success: false, error: error.message }, { status: 500 })
  }
}
