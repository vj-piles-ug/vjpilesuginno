import { NextResponse } from "next/server"

const FIREBASE_DB_URL = "https://vj-pilesug-default-rtdb.firebaseio.com"

async function safeFetch(url: string): Promise<any> {
  try {
    const res = await fetch(url, { cache: "no-store" })
    if (!res.ok) return null
    const text = await res.text()
    if (text === "null" || text === "") return null
    const data = JSON.parse(text)
    if (data && typeof data === "object" && !data.error) return data
    return null
  } catch {
    return null
  }
}

export async function GET() {
  try {
    // Fetch paymentLogs, pendingOrders, and subscriptions in parallel
    const [logsData, pendingData, subsData] = await Promise.all([
      safeFetch(`${FIREBASE_DB_URL}/paymentLogs.json`),
      safeFetch(`${FIREBASE_DB_URL}/pendingOrders.json`),
      safeFetch(`${FIREBASE_DB_URL}/subscriptions.json`),
    ])

    const paymentsMap: Record<string, any> = {}

    // 1. Subscriptions = confirmed successful payments
    if (subsData) {
      for (const [userId, value] of Object.entries(subsData)) {
        const v = value as any
        const trackingId = v.orderTrackingId || `sub-${userId}`
        paymentsMap[trackingId] = {
          id: trackingId,
          userId,
          email: v.email || "",
          phoneNumber: v.paymentPhone || v.phoneNumber || "",
          firstName: v.firstName || v.displayName || "",
          lastName: v.lastName || "",
          amount: v.amount || 0,
          planId: v.planId || "",
          planName: v.planName || "",
          days: v.days || 0,
          orderTrackingId: trackingId,
          merchantReference: v.merchantReference || "",
          status: "successful",
          paymentMethod: v.paymentMethod || "",
          confirmationCode: v.confirmationCode || "",
          createdAt: v.createdAt || v.startDate || "",
          updatedAt: v.createdAt || "",
        }
      }
    }

    // 2. Payment logs (may override or add new entries)
    if (logsData) {
      for (const [key, value] of Object.entries(logsData)) {
        const v = value as any
        // Only add if not already from subscriptions, or update with more info
        if (!paymentsMap[key]) {
          paymentsMap[key] = {
            id: key,
            userId: v.userId || "",
            email: v.email || "",
            phoneNumber: v.phoneNumber || v.phone_number || v.payment_account || "",
            firstName: v.firstName || v.first_name || "",
            lastName: v.lastName || v.last_name || "",
            amount: v.amount || 0,
            planId: v.planId || "",
            planName: v.planName || "",
            days: v.days || 0,
            orderTrackingId: v.orderTrackingId || key,
            merchantReference: v.merchantReference || "",
            status: normalizeStatus(v.status),
            paymentMethod: v.paymentMethod || v.payment_method || "",
            confirmationCode: v.confirmationCode || v.confirmation_code || "",
            createdAt: v.createdAt || v.created_at || "",
            updatedAt: v.updatedAt || v.updated_at || "",
          }
        }
      }
    }

    // 3. Pending orders not yet in other sources
    if (pendingData) {
      for (const [key, value] of Object.entries(pendingData)) {
        if (!paymentsMap[key]) {
          const v = value as any
          paymentsMap[key] = {
            id: key,
            userId: v.userId || "",
            email: v.email || "",
            phoneNumber: v.phoneNumber || v.phone_number || "",
            firstName: v.firstName || v.first_name || "",
            lastName: v.lastName || v.last_name || "",
            amount: v.amount || 0,
            planId: v.planId || "",
            planName: v.planName || "",
            days: v.days || 0,
            orderTrackingId: key,
            merchantReference: "",
            status: "pending",
            paymentMethod: "",
            confirmationCode: "",
            createdAt: v.createdAt || "",
            updatedAt: "",
          }
        }
      }
    }

    const payments = Object.values(paymentsMap)
    return NextResponse.json({ payments })
  } catch (error: any) {
    console.error("[payment-log GET] error:", error)
    return NextResponse.json({ payments: [], error: error.message }, { status: 500 })
  }
}

function normalizeStatus(status: string | undefined): string {
  if (!status) return "pending"
  const s = status.toLowerCase()
  if (s === "completed" || s === "successful" || s === "success") return "successful"
  if (s === "failed" || s === "fail") return "failed"
  if (s === "reversed") return "reversed"
  return "pending"
}

export async function POST(request: Request) {
  try {
    const body = await request.json()
    const { orderTrackingId, status, phoneNumber, paymentMethod, confirmationCode } = body

    if (!orderTrackingId) {
      return NextResponse.json({ error: "Missing orderTrackingId" }, { status: 400 })
    }

    await fetch(`${FIREBASE_DB_URL}/paymentLogs/${orderTrackingId}.json`, {
      method: "PATCH",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        status: status || "unknown",
        phoneNumber: phoneNumber || "",
        paymentMethod: paymentMethod || "",
        confirmationCode: confirmationCode || "",
        updatedAt: new Date().toISOString(),
      }),
    })

    return NextResponse.json({ success: true })
  } catch (error: any) {
    return NextResponse.json({ error: error.message }, { status: 500 })
  }
}
