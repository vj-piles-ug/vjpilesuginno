import { NextResponse } from "next/server"

const PESAPAL_CONSUMER_KEY = process.env.PESAPAL_CONSUMER_KEY || "vTibxQeLvkiJ0aBZQk3Ci5gL2BUx1dP0"
const PESAPAL_CONSUMER_SECRET = process.env.PESAPAL_CONSUMER_SECRET || "kdwghPqFeT5dxwnhrpVccfY9B7o="
const PESAPAL_API_URL = "https://pay.pesapal.com/v3"

async function getAuthToken() {
  const response = await fetch(`${PESAPAL_API_URL}/api/Auth/RequestToken`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      Accept: "application/json",
    },
    body: JSON.stringify({
      consumer_key: PESAPAL_CONSUMER_KEY,
      consumer_secret: PESAPAL_CONSUMER_SECRET,
    }),
  })
  const data = await response.json()
  return data.token
}

// Register IPN URL (used by submit-order)
export async function POST(request: Request) {
  try {
    const body = await request.json()
    const { callbackUrl } = body

    const token = await getAuthToken()
    if (!token) {
      return NextResponse.json({ success: false, error: "Failed to authenticate" }, { status: 401 })
    }

    const response = await fetch(`${PESAPAL_API_URL}/api/URLSetup/RegisterIPN`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        Accept: "application/json",
        Authorization: `Bearer ${token}`,
      },
      body: JSON.stringify({
        url: callbackUrl,
        ipn_notification_type: "GET",
      }),
    })

    const data = await response.json()

    if (data.ipn_id) {
      return NextResponse.json({ success: true, ipn_id: data.ipn_id, url: data.url })
    } else {
      return NextResponse.json({ success: false, error: data.error?.message || "Failed to register IPN" }, { status: 400 })
    }
  } catch (error: any) {
    return NextResponse.json({ success: false, error: error.message }, { status: 500 })
  }
}

const FIREBASE_DB_URL = "https://vj-pilesug-default-rtdb.firebaseio.com"

// Handle IPN callback from PesaPal (GET request with query params)
// PesaPal docs: IPN call has OrderTrackingId, OrderMerchantReference, OrderNotificationType=IPNCHANGE
// We must respond with: { orderNotificationType, orderTrackingId, orderMerchantReference, status: 200 }
export async function GET(request: Request) {
  const { searchParams } = new URL(request.url)
  const orderTrackingId = searchParams.get("OrderTrackingId") || ""
  const orderMerchantReference = searchParams.get("OrderMerchantReference") || ""
  const orderNotificationType = searchParams.get("OrderNotificationType") || ""

  try {
    if (orderTrackingId && orderNotificationType === "IPNCHANGE") {
      const token = await getAuthToken()
      if (token) {
        // Get payment status from PesaPal
        const statusResponse = await fetch(
          `${PESAPAL_API_URL}/api/Transactions/GetTransactionStatus?orderTrackingId=${orderTrackingId}`,
          {
            headers: {
              "Content-Type": "application/json",
              Accept: "application/json",
              Authorization: `Bearer ${token}`,
            },
          }
        )
        const statusData = await statusResponse.json()
        const statusCode = typeof statusData.status_code === "number" ? statusData.status_code : -1

        // Try to update Firebase (may fail with 401 in some runtimes, that's OK)
        // The client-side polling is the primary activation path
        try {
          // Look up pending order
          const pendingRes = await fetch(`${FIREBASE_DB_URL}/pendingOrders/${orderTrackingId}.json`)
          const pendingData = pendingRes.ok ? await pendingRes.json() : null

          const finalStatus = statusCode === 1 ? "successful" : statusCode === 2 ? "failed" : statusCode === 3 ? "reversed" : "pending"

          // Update payment log
          await fetch(`${FIREBASE_DB_URL}/paymentLogs/${orderTrackingId}.json`, {
            method: "PATCH",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({
              status: finalStatus,
              phoneNumber: statusData.payment_account || "",
              paymentMethod: statusData.payment_method || "",
              confirmationCode: statusData.confirmation_code || "",
              updatedAt: new Date().toISOString(),
            }),
          })

          // If completed and we have pending order data, activate subscription
          if (statusCode === 1 && pendingData && pendingData.userId) {
            const startDate = new Date()
            const endDate = new Date()
            endDate.setDate(endDate.getDate() + (pendingData.days || 1))

            await fetch(`${FIREBASE_DB_URL}/subscriptions/${pendingData.userId}.json`, {
              method: "PUT",
              headers: { "Content-Type": "application/json" },
              body: JSON.stringify({
                planId: pendingData.planId || "unknown",
                planName: pendingData.planName || "Subscription",
                amount: pendingData.amount || statusData.amount || 0,
                days: pendingData.days || 1,
                orderTrackingId,
                merchantReference: orderMerchantReference || "",
                confirmationCode: statusData.confirmation_code || "",
                paymentMethod: statusData.payment_method || "PesaPal",
                startDate: startDate.toISOString(),
                endDate: endDate.toISOString(),
                active: true,
                createdAt: new Date().toISOString(),
              }),
            })

            // Clean up pending order
            await fetch(`${FIREBASE_DB_URL}/pendingOrders/${orderTrackingId}.json`, {
              method: "DELETE",
            })
          }
        } catch {
          // Firebase REST writes may fail - client-side polling handles activation
        }
      }
    }
  } catch {
    // Don't let errors prevent the IPN response
  }

  // Always respond to PesaPal with status 200 per their docs
  return NextResponse.json({
    orderNotificationType: orderNotificationType || "IPNCHANGE",
    orderTrackingId,
    orderMerchantReference,
    status: 200,
  })
}
