"use client"

import type React from "react"
import { createContext, useContext, useEffect, useState } from "react"
import { database } from "./firebase"
import { ref, get, set, update, onValue } from "firebase/database"
import { useAuth } from "./auth-context"

export interface SubscriptionPlan {
  id: string
  name: string
  duration: string
  price: number
  days: number
}

export const SUBSCRIPTION_PLANS: SubscriptionPlan[] = [
  { id: "2days", name: "2 Days Pass", duration: "2 Days", price: 5000, days: 2 },
  { id: "1week", name: "1 Week Pass", duration: "1 Week", price: 15000, days: 7 },
  { id: "2weeks", name: "2 Weeks Pass", duration: "2 Weeks", price: 25000, days: 14 },
  { id: "1month", name: "1 Month Pass", duration: "1 Month", price: 40000, days: 30 },
]

interface Subscription {
  planId: string
  startDate: string
  endDate: string
  active: boolean
}

interface SubscriptionContextType {
  subscription: Subscription | null
  hasActiveSubscription: boolean
  loading: boolean
  refreshSubscription: () => Promise<void>
}

const SubscriptionContext = createContext<SubscriptionContextType | undefined>(undefined)

export function SubscriptionProvider({ children }: { children: React.ReactNode }) {
  const { user } = useAuth()
  const [subscription, setSubscription] = useState<Subscription | null>(null)
  const [loading, setLoading] = useState(true)

  const checkSubscription = async () => {
    if (!user) {
      setSubscription(null)
      setLoading(false)
      return
    }

    try {
      const subscriptionRef = ref(database, `subscriptions/${user.uid}`)
      const snapshot = await get(subscriptionRef)

      if (snapshot.exists()) {
        const data = snapshot.val()
        const endDate = new Date(data.endDate)
        const now = new Date()

        // Check if subscription is still active
        if (endDate > now) {
          setSubscription(data)
        } else {
          setSubscription({ ...data, active: false })
        }
      } else {
        setSubscription(null)
      }
    } catch (error) {
      console.error("Error checking subscription:", error)
      setSubscription(null)
    } finally {
      setLoading(false)
    }
  }

  // Silent global payment poller: runs on every page, checks every 2 seconds
  // for any pending payment in localStorage. When PesaPal confirms success,
  // activates the subscription automatically. Cleans up on failure/reversal.
  // Survives page refreshes, modal closes, and navigation.
  useEffect(() => {
    if (!user) return

    let interval: ReturnType<typeof setInterval> | null = null
    let active = true
    let polling = false // guard against overlapping requests

    const pollPending = async () => {
      if (!active || polling) return
      polling = true

      try {
        await doPollPending()
      } finally {
        polling = false
      }
    }

    const doPollPending = async () => {
      const pendingStr = localStorage.getItem("pendingSubscription")
      if (!pendingStr) {
        // No pending payment, stop polling
        if (interval) { clearInterval(interval); interval = null }
        return
      }

      let pending: any
      try { pending = JSON.parse(pendingStr) } catch { localStorage.removeItem("pendingSubscription"); return }
      if (!pending.orderTrackingId) { localStorage.removeItem("pendingSubscription"); return }

      // Check if already has active subscription (maybe activated by another tab or IPN)
      const subRef = ref(database, `subscriptions/${user.uid}`)
      try {
        const subSnap = await get(subRef)
        if (subSnap.exists()) {
          const data = subSnap.val()
          const endDate = new Date(data.endDate)
          if (endDate > new Date() && data.active) {
            localStorage.removeItem("pendingSubscription")
            if (interval) { clearInterval(interval); interval = null }
            return
          }
        }
      } catch {}

      // Poll PesaPal for transaction status
      try {
        const res = await fetch(
          `/api/pesapal/transaction-status?orderTrackingId=${pending.orderTrackingId}&_t=${Date.now()}`,
          { cache: "no-store", headers: { "Cache-Control": "no-cache" } }
        )
        if (!res.ok) return
        const data = await res.json()
        const statusCode = data.status_code ?? data.statusCode ?? -1

        if (statusCode === 1) {
          // COMPLETED - activate subscription silently
          const startDate = new Date()
          const endDate = new Date()
          endDate.setDate(endDate.getDate() + (pending.days || 1))

          await Promise.all([
            set(ref(database, `subscriptions/${user.uid}`), {
              planId: pending.planId,
              planName: pending.planName,
              amount: data.amount || pending.amount,
              days: pending.days || 1,
              orderTrackingId: pending.orderTrackingId,
              merchantReference: pending.merchantReference || "",
              confirmationCode: data.confirmation_code || "",
              paymentMethod: data.payment_method || "PesaPal",
              phoneNumber: data.payment_account || "",
              startDate: startDate.toISOString(),
              endDate: endDate.toISOString(),
              active: true,
              createdAt: new Date().toISOString(),
            }),
            update(ref(database, `paymentLogs/${pending.orderTrackingId.replace(/[.#$/\[\]]/g, "_")}`), {
              phoneNumber: data.payment_account || "",
              amount: data.amount || pending.amount,
              status: "successful",
              paymentMethod: data.payment_method || "PesaPal",
              confirmationCode: data.confirmation_code || "",
              updatedAt: new Date().toISOString(),
            }),
          ])

          localStorage.removeItem("pendingSubscription")
          if (interval) { clearInterval(interval); interval = null }
        } else if (statusCode === 2 || statusCode === 3) {
          // FAILED or REVERSED - log to Firebase for admin visibility, then clean up
          try {
            await update(ref(database, `paymentLogs/${pending.orderTrackingId.replace(/[.#$/\[\]]/g, "_")}`), {
              phoneNumber: data.payment_account || "",
              amount: data.amount || pending.amount || 0,
              status: statusCode === 3 ? "reversed" : "failed",
              paymentMethod: data.payment_method || "",
              confirmationCode: data.confirmation_code || "",
              updatedAt: new Date().toISOString(),
            })
          } catch {}
          localStorage.removeItem("pendingSubscription")
          if (interval) { clearInterval(interval); interval = null }
        }
        // statusCode 0 or -1 = still pending, keep polling
      } catch {
        // Network error, keep trying next interval
      }
    }

    // Start polling every 3 seconds (avoids hammering the API)
    pollPending()
    interval = setInterval(pollPending, 3000)

    return () => {
      active = false
      if (interval) clearInterval(interval)
    }
  }, [user])

  // Use real-time listener to immediately unlock when subscription is activated
  useEffect(() => {
    if (!user) {
      setSubscription(null)
      setLoading(false)
      return
    }

    setLoading(true)

    const subscriptionRef = ref(database, `subscriptions/${user.uid}`)
    
    // Set up real-time listener
    const unsubscribe = onValue(subscriptionRef, (snapshot) => {
      if (snapshot.exists()) {
        const data = snapshot.val()
        const endDate = new Date(data.endDate)
        const now = new Date()

        // Check if subscription is still active
        if (endDate > now) {
          setSubscription({ ...data, active: true })
        } else {
          setSubscription({ ...data, active: false })
        }
      } else {
        setSubscription(null)
      }
      setLoading(false)
    }, () => {
      setSubscription(null)
      setLoading(false)
    })

    // Cleanup listener on unmount
    return () => unsubscribe()
  }, [user])

  const hasActiveSubscription = subscription?.active === true

  return (
    <SubscriptionContext.Provider
      value={{ subscription, hasActiveSubscription, loading, refreshSubscription: checkSubscription }}
    >
      {children}
    </SubscriptionContext.Provider>
  )
}

export function useSubscription() {
  const context = useContext(SubscriptionContext)
  if (context === undefined) {
    throw new Error("useSubscription must be used within a SubscriptionProvider")
  }
  return context
}
